\section{Orações}
Esta é uma lista com as orações abreviadas ao longo deste documento, que também são tipicamente usadas no Rosário ou Terço. São orações da Igreja presentes em diversos lugares da tradição, desde as Sagradas Escrituras até aparições privadas.



\subsection{Credo dos Apóstolos}\label{creio}
\begin{verses}

Creio em Deus, Pai todo-poderoso, Criador do Céu e da Terra\\!

E em Jesus Cristo, seu único Filho, nosso Senho\\
que foi concebido pelo poder do Espírito Santo\\
nasceu da Virgem Maria;\\
padeceu sob Pôncio Pilatos,\\
foi crucificado, morto e sepultado;\\
desceu à mansão dos mortos;\\
ressuscitou ao terceiro dia;\\
subiu aos Céus;\\
está sentado à direita de Deus Pai todo-poderoso,\\
de onde há-de vir a julgar os vivos e os mortos\\!

Creio no Espírito Santo; \\
na santa Igreja Católica; \\
na comunhão dos Santos; \\
na remissão dos pecados; \\
na ressurreição da carne; \\
e na vida eterna.\\

\end{verses}

\subsection{Sinal da Cruz}\label{sinal-da-cruz}
\begin{verses}
Em nome do Pai, do Filho e do Espírito Santo. Amém. \footnote{Deve-se levar a mão direita primeiramente à testa, pois foi na testa que, no dia do nosso Batismo, recebemos pela primeira vez o Sinal da Cruz de Cristo. Depois, a mão deve ser dirigida na altura do peito, em seguida, a mão deve ser levada ao ombro esquerdo, depois ao ombro direito. No momento do "amém", não deve ser dado o beijo na mão, pois o próprio "Amém" confirma o que se acabou de rezar.}
\end{verses}

\subsection{Persignação}\label{persignacao}
\begin{verses}
  Pelo sinal da Santa Cruz (\dag cruz na testa)\footnote{Com a mão direita fazer o sinal da cruz, de cima para baixo e da esquerda para a direita… },\\  
  livrai-nos Deus, Nosso Senhor († cruz na boca),\\ 
  de nossos inimigos (\dag cruz no peito) .

  Amém.
\end{verses}

\subsection{Pai-Nosso}\label{pai-nosso}

\begin{verses}

Pai Nosso que estais nos Céus,\\
santificado seja o vosso Nome,\\
venha a nós o vosso Reino,\\
seja feita a vossa vontade\\
assim na terra como no Céu.\\
O pão nosso de cada dia nos dai hoje,\\
perdoai-nos as nossas ofensas\\
assim como nós perdoamos\\
a quem nos tem ofendido,\\
e não nos deixeis cair em tentação,\\
mas livrai-nos do Mal.\\

\end{verses}

\subsection{Ave-Maria}\label{ave-maria}

\begin{verses}
Ave Maria, cheia de graça, \\
o Senhor é convosco, \\
bendita sois vós entre as mulheres \\
e bendito é o fruto do vosso ventre, Jesus. \\
Santa Maria, Mãe de Deus, \\
rogai por nós pecadores, \\
agora e na hora da nossa morte. Ámen\\
\end{verses}
\subsection{Salve-Rainha}\label{salve-rainha}

\begin{verses}
Salve, Rainha, \\
mãe de misericórdia, \\
vida, doçura, esperança nossa, salve! \\
A Vós bradamos, \\
os degredados filhos de Eva. \\
A Vós suspiramos, gemendo e chorando \\
neste vale de lágrimas. \\
Eia, pois, advogada nossa, \\
esses Vossos olhos misericordiosos \\
a nós volvei. \\
E, depois deste desterro, \\
nos mostrai Jesus, bendito fruto \\
do Vosso ventre. \\
Ó clemente, ó piedosa, \\
ó doce Virgem Maria. \\
Rogai por nós, Santa Mãe de Deus, \\
para que sejamos dignos das promessas de Cristo.\\
\end{verses}

\subsection{Glória}\label{gloria}
\begin{verses}
  Glória ao Pai, ao Filho e ao Espírito Santo. \\Como era no princípio, agora e sempre. Amém.
\end{verses}

\subsection{Ó Meu Jesus}\label{oh-meu-jesus}
\begin{verses}
  Ó meu Jesus,\\
   perdoai-nos, livrai-nos do fogo do inferno,\\ 
   levai as almas todas para o Céu,\\
   principalmente, as que mais precisarem.
\end{verses}

\subsection{Jaculatórias}\label{oracoes-jaculatorias}

\subsubsection*{Jaculatória da Imaculada Conceição}\label{jaculatoria-da-imaculada-conceição}

\begin{verses}
Ó Maria, concebida sem pecado, rogai por nós que recorremos a Vós.
\end{verses}

\subsubsection*{Jaculatória do Coração de Jesus.}\label{jaculatoria-do-coracao-de-jesus}


\begin{verses}
Jesus, manso e humilde de coração, fazei o nosso coração semelhante ao Vosso.
\end{verses}

\subsubsection*{Súplica ao Santo de Devoção}\label{suplica-ao-santo-de-devocao}

\begin{verses}
\textbf{[Nome do santo]}, rogai por nós!
\end{verses}

\subsubsection*{Jaculatória de Cada Mistério.}\label{jaculatoria-de-cada-mistério}

\begin{verses}
  Que a Graça do mistério [citar o mistério] desça em nossas almas.
\end{verses}
